Before deploying make sure you've created your secrets:

  kubectl create secret generic {{ .Values.secretName }} \
    --from-literal=HASH_SECRET="..." \
    --from-literal=JWT_SECRET="..." \
    --from-literal=ENCRYPTION_KEY="$(openssl rand -base64 32)"
    --from-literal=INTERNAL_API_KEY="$(base64 -i - <<< internal:password)"


kubectl create secret docker-registry {{ .Values.imagePullSecret }} \
  --docker-server=10.10.10.245:5000 \
  --docker-username=<user> \
  --docker-password=<password> \
  --namespace=<your-namespace>
